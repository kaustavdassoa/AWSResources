#!/bin/bash
sudo su
yum update -y
sudo yum upgrade
sudo yum install git -y
mkdir /home/ec2-user/download
chmod 777 /home/ec2-user/download 
cd /home/ec2-user/download
git clone https://github.com/kaustavdassoa/NodeJS-Dev.git
cd /home/ec2-user/download/NodeJS-Dev/blockchain-app/
curl --silent --location https://rpm.nodesource.com/setup_6.x | sudo bash
sudo yum -y install nodejs
sudo npm install forever -g
cd /home/ec2-user/download/NodeJS-Dev/blockchain-app
npm install 
cd /home/ec2-user/download/NodeJS-Dev/blockchain-app/server
forever start server.js
forever list > /home/ec2-user/forever-log.log
